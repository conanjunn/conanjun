git init  ——初始化一个git项目

git add myFireName/*   ——添加修改  *代表所有文件
git commit  ——提交修改
git commit -a  ——等于git add * +  git commit
git commit -m massage  ——写入提交信息，并提交

git status ——显示当前状态
git diff ——查看改动
git log ——查看日志
git log --pretty=oneline ——打印在一行

git reset --hard HEAD^ ——后退1步
git reset --hard HEAD^^  ——后退2步
git reset --hard HEA~100 ——后退100步
git reset --hard commitId ——后退到commitId
git reflog ——查看命令历史
git checkout -- fileName ——撤销修改
git rm ——删除文件

git remote add origin git@server-name:path/repo-name.git;  ——关联远程仓库
git remote  ——查看远程库的信息
git remote -v  ——查看远程库的信息（更详细）
git push (remote)(branch)  ——向远程服务器推送代码
git clone ssh://user@domain.com/repo.git  --从远程仓库克隆

git pull  ——抓取远程的最新提交

git branch dev ——创建dev分支
git branch -d dev ——删除dev分支
git branch -D dev ——强行删除dev分支，即使它未被合并
git checkout dev ——切换到Dev分支
git checkout -b dev  ——上面两条的简写
git branch ——打印所有分支，当前分支前面会标一个*号
git merge --no-ff dev ——禁用“Fast forward”，创建一个新的commit

git stash 存储工作区
git stash list 查看存储列表
git stash apply stash@{0} ——恢复存储（不删除存储）
git stash pop stash@{0} ——恢复存储（删除存储）


git tag  打标签
git tag  查看所有标签
git show tagname 查看标签信息
git tag -d v0.1  删除标签
命令git push origin tagname可以推送一个本地标签；
命令git push origin --tags可以推送全部未推送过的本地标签；
命令git tag -d tagname可以删除一个本地标签；
命令git push origin :refs/tags/tagname可以删除一个远程标签。
